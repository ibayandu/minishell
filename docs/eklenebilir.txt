char	*ft_strndup(const char *s1, size_t n)
{
	char	*newstr;
	size_t	strlen;

	strlen = ft_strlen(s1);
	if (n > strlen)
		n = strlen;
	newstr = ft_calloc(n + 1, sizeof(char));
	if (!newstr)
		return (NULL);
	ft_memcpy(newstr, s1, n);
	return (newstr);
}
